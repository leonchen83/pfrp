    A implementation of RFC6901.

    A implementation of RFC6901. Provide a convenient way to access json object using the path which like file path.

    RFC6901 example
    Example: 

    1. // For example, given the JSON document
{
  "foo": ["bar", "baz"],
  "": 0,
  "a/b": 1,
  "c%d": 2,
  "e^f": 3,
  "g|h": 4,
  "i\\j": 5,
  "k\"l": 6,
  " ": 7,
  "m~n": 8
}

// The following JSON strings evaluate to the accompanying values:

""           // the whole document
"/foo"       ["bar", "baz"]
"/foo/0"     "bar"
"/"          0
"/a~1b"      1
"/c%d"       2
"/e^f"       3
"/g|h"       4
"/i\\j"      5
"/k\"l"      6
"/ "         7
"/m~0n"      8

      usage

      val json =
  """
    |{
    |  "store": {
    |    "book": [
    |      { "category": "reference",
    |        "author": "Nigel Rees",
    |        "title": "Sayings of the Century",
    |        "price": 8.95
    |      },
    |      { "category": "fiction",
    |        "author": "Evelyn Waugh",
    |        "title": "Sword of Honour",
    |        "price": 12.99
    |      },
    |      { "category": "fiction",
    |        "author": "Herman Melville",
    |        "title": "Moby Dick",
    |        "isbn": "0-553-21311-3",
    |        "price": 8.99
    |      },
    |      { "category": "fiction",
    |        "author": "J. R. R. Tolkien",
    |        "title": "The Lord of the Rings",
    |        "isbn": "0-395-19395-8",
    |        "price": 22.99
    |      }
    |    ],
    |    "bicycle": {
    |      "color": "red",
    |      "price": 19.95
    |    }
    |  }
    |}
  """.stripMargin
val jp = JSONPointer(json)
val value = jp.path("/store/book/0-2/isbn")
